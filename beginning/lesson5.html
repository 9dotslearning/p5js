12. Make Your Own Project

Tinker

To make and save your own project, you'll work in the p5.js Web Editor. Click this link to open the p5.js Web Editor in another tab. A new project opens. Select all the code and press Delete or Backspace to delete it. You can copy and paste one of the code examples below to use as a starter project or simply use the examples as a reference for making a completely new project of your own. The first example creates falling rectangles, and the second one creates rising ellipses. Change the code to make an animation of your own, similar to either of the GIFs at the beginning of this lesson. Have fun! üëè 

/* *** FALLING RECTANGLES *** */

let c = 50;
let y = 0;

function setup() {
  createCanvas(250, 200);
  colorMode(HSB);
  background(30, 10, 30);
  frameRate(10);
}

function draw() {
  fill(c, 100, 100);
  rect(4, y, 20, 20);
  rect(34, y, 20, 20);
  y = y + 8;

  if (y > 200) c = 325;
  if (y > 200) y = 0;
}
          
/* *** RISING ELLIPSES *** */

let x = 125;
let y = 200;

function setup() {
  createCanvas(250, 200);
  colorMode(HSB);
  background(30, 10, 30);
}

function draw() {
  fill(0, 100, 100);
  ellipse(x, y, 20, 20);
  y = y - 12;

  x = x + random(-3, 3);

  if (y < 0) x = random(200);
  if (y < 0) y = 200;
}
